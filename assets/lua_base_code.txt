        bulgind = true
        
        local x = e.x
        local y = e.y + 1
        local z = e.z
        
        local x_face = 1
        local y_face = 2
        local z_face = 3
        
       for i = 1, #positions do
         --threadpool:wait(0.0000000000001)
         Block:setBlockAll(x + positions[i][1] ,y + positions[i][3],z+ positions[i][2],blocks[positions[i][4] + 1][1],blocks[positions[i][4]+ 1][2])
       end

end
